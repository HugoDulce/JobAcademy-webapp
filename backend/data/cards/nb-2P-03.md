from sklearn.model_selection import GridSearchCV
from sklearn.naive_bayes import GaussianNB

param_grid = {
    'var_smoothing': np.logspace(-10, -1, 10)
}

grid = GridSearchCV(
    GaussianNB(), 
    param_grid, 
    cv=5, 
    scoring='f1_weighted'
)

grid.fit(X_train, y_train)
print(f"Best smoothing: {grid.best_params_}")
print(f"Best F1: {grid.best_score_:.3f}")


Why logspace: var_smoothing ranges exponentially (1e-10 to 1e-1).

Why F1: Better than accuracy for imbalanced classes.
